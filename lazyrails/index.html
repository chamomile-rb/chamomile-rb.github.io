Smart Introspection

One rails runner call replaces 3+ separate CLI commands. Structured JSON, not fragile text parsing.

introspect_runner.rb 
# Routes via ActionDispatch::Routing
data[:routes] = Rails.application.routes.routes.filter_map do |route|
  w = ActionDispatch::Routing::RouteWrapper.new(route)
  next if w.internal?
  { verb: w.verb, path: w.path, action: w.endpoint, name: w.name }
end

# Schema columns directly from the connection
conn = ActiveRecord::Base.connection
data[:tables] = conn.tables.sort.to_h do |table|
  cols = conn.columns(table).map do |c|
    { name: c.name, type: c.type, null: c.null, default: c.default }
  end
  [table, cols]
end

# Models with associations and validations
data[:models] = ActiveRecord::Base.descendants.filter_map do |model|
  next if model.abstract_class? || !model.table_exists?
  assocs = model.reflect_on_all_associations.map { |a|
    { macro: a.macro, name: a.name, class_name: a.class_name }
  }
  { name: model.name, table: model.table_name, associations: assocs }
end

Clean Data Modeling

Immutable structs with Data.define. Every piece of data has a clear shape.

Route 
Route = Data.define(:verb, :path, :action, :name, :engine) do
  def to_s
    "#{verb.ljust(7)} #{path.ljust(40)} #{action}"
  end
end
Migration 
Migration = Data.define(
  :status, :version, :name, :file_path, :database
) do
  def up? = status == "up"
  def down? = status == "down"

  def to_s
    arrow = up? ? "\u2191" : "\u2193"
    "#{arrow} #{version}  #{name}"
  end
end

Subprocess Mastery

Sophisticated server lifecycle management. Process groups, Bundler isolation, intelligent state detection.

server_manager.rb 
class ServerManager
  MAX_LOG_LINES = 2000
  KILL_TIMEOUT = 5

  def start
    @state = :starting
    @thread = Thread.new do
      env = { "DISABLE_SPRING" => "1", "PORT" => @port.to_s }
      cmd = @uses_bin_dev ? "bin/dev" : "#{@project.bin_rails} server"

      Bundler.with_unbundled_env do
        Open3.popen2e(env, cmd, chdir: @project.dir, pgroup: true) do |stdin, out, wait|
          stdin.close
          out.each_line do |line|
            @mutex.synchronize do
              @log_lines << line
              @log_lines.shift if @log_lines.size > MAX_LOG_LINES
              @state = :running if server_ready?(line)
            end
          end
        end
      end
    end
  end

  private

  def server_ready?(line)
    line.include?("Listening on") ||  # Puma
    line.include?("listening on") ||  # Falcon, Vite
    line.include?("Ctrl-C to stop")   # WEBrick, Thin
  end
end

Safety Built In

Destructive operations require confirmation proportional to their risk. Every command is logged with undo support.

confirmation.rb 
class Confirmation
  # Red: type panel name to confirm
  RED_PATTERNS = %w[db:drop db:reset db:seed:replant]

  # Yellow: y/n prompt
  YELLOW_PATTERNS = %w[db:rollback db:migrate:down destroy]

  def self.detect_tier(cmd)
    cmd = cmd.join(" ") if cmd.is_a?(Array)
    return :red if RED_PATTERNS.any? { |p| cmd.include?(p) }
    return :yellow if YELLOW_PATTERNS.any? { |p| cmd.include?(p) }
    :green
  end
end
command_log.rb — undo 
def reverse_command(entry)
  case entry.command
  when /rails generate (.+)/
    entry.command.sub("generate", "destroy")
  when /db:migrate\b(?!:)/
    "bin/rails db:rollback"
  when /db:migrate:up VERSION=(\d+)/
    version = Regexp.last_match(1)
    "bin/rails db:migrate:down VERSION=#{version}"
  end
end

Rich Detail Views

Drill into any item for full context. Routes show verb coloring and controller resolution.

VERB_COLORS = {
  "GET"    => "#04b575",
  "POST"   => "#5b9bd5",
  "PUT"    => "#e5c07b",
  "PATCH"  => "#e5c07b",
  "DELETE" => "#ff6347",
}

def self.render_detail(route, project_dir, width:)
  lines = ["#{route.verb} #{route.path}"]
  lines << ("=" * [width - 4, 40].min)
  lines << "Action:     #{route.action}"

  if route.action.include?("#")
    controller, action = route.action.split("#", 2)
    path = "app/controllers/#{controller}_controller.rb"
    content = safe_read(File.join(project_dir, path))
    if content&.include?("def #{action}")
      lines << "Method:     def #{action} \u2713"
    end
  end
  lines.join("\n")
end
